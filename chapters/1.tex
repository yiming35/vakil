\section{Just enough category theory to be dangerous}
\subsection{Categories and functors}
\begin{exer}[1.1.A] Groupoids.
    \begin{enumerate}[(a)]
        \item Make sense of the statement ``a group is a groupoid with one element.''
        \begin{soln}
            Let $\mathcal{G}$ be a one element groupoid, call the single element $x$. Then by definition the identity morphism is in $\hom_\mathcal{G}(x,x)$. We will take this to be the identity element of the group $\hom_\mathcal{G}(x,x)$. Any two morphisms are composable, which gives the group composition, and by definition of groupoid any morphism is invertible, yielding the group inverse.

            Conversely, if we have a group $G$ we can make a category with one object whose morphisms are the elements of $G$, and whose identity, composition, and inverses are given by the group structure. This yields a groupoid.
        \end{soln}
        \item Give an example of a groupoid that is not a group.
        \begin{soln}
        A category with more than one object, and only the identity morphisms. 
        \end{soln}
    \end{enumerate}
\end{exer}

\begin{exer}[1.1.B] If $A$ is an object in a category $\mathcal{C}$, show that the invertible elements of $\hom_{\mathcal{C}}(A,A)$ form a group, called the automorphism group. What are the automorphism groups in $\catname{Set}$ and $\catname{Vect}_k$?
    \begin{proof}
    This is exactly the same as part 1 above. The identity morphism serves as the identity, the inverse morphisms yield the group inverses, and the category axioms yield composition. 
    
    The automorphism groups in $\catname{Set}$ are the symmetric groups, and the automorphism groups in $\catname{Vect}_k$ are the general linear groups.
    \end{proof}
\end{exer}

\begin{exer}[1.1.C.] Fix a field $k$. Let $(-)^{\vee\vee}$ be the double dual functor from the category of finite-dimensional $k$-vector spaces to itself. Show that $(-)^{\vee\vee}$ is naturally isomorphic to the identity functor.
    \begin{proof}
    For a fixed vector space $V$, define the map $\alpha_V: V \to V^{\vee\vee} $ by $v \mapsto (\phi\mapsto \phi(v))$. This is injective because if $\phi \mapsto \phi(v)$ is the zero map for every $\phi$, then $v=0$, so by finite-dimensionality $\alpha_V$ is an isomorphism. And it remains to check that for any map of vector spaces $f: V\to W$ and $v \in V$, that 
    \begin{align*}
        \alpha_W(f(v)) &= (\phi \mapsto \phi(f(v))) \\
        &= (\psi \mapsto \psi(v)) \circ f^\vee \\
        &= f^{\vee\vee}(\psi \mapsto \psi(v)) \\
        &= f^{\vee\vee}(\alpha_V(v))
    \end{align*}
    which completes the proof.
    \end{proof}
\end{exer}

\begin{exer}[1.1.D] Let $\mathcal{V}$ be the category whose objects are the $k$-vector spaces $k^n$ for $n\geq 0$, and whose morphisms are linear transformations. There is a functor $F$ from $\mathcal{V}$ to the category of f.d. $k$-vector spaces sending each vector space to itself. Show that $F$ is an equivalence of categories.
    \begin{proof}
    Following Vakil's remark about ignoring set-theoretic conditions, choose a basis for each finite dimensional $k$-vector space. For any vector space of the form $k^n$, choose the standard basis $\{e_i\}$.
    
    Then define a functor $G$ from f.d. $k$-vector spaces to $\mathcal{V}$ by sending an $n$-dimensional $k$-vector space to $k^n$. For a morphism $\phi$ from $V$ to an $m$-dimensional vector space $W$, let $\{v_1, ..., v_n\}$ be the chosen basis for $V$ and $\{w_1, ..., w_m\}$ the chosen basis for $W$. Then define $G\phi$ to be the matrix whose $(i,j)$th entry is the coefficient of $w_j$ in $\phi(v_i)$. 
    
    Define a natural transformation $\alpha$ from $FG$ to the identity functor where $\alpha_V: FGV \to V$ sends $FGV = k^n$ to $V$ by mapping the standard basis vector $e_i$ of $FGV$ to the chosen $i$th basis vector $v_i$ of $V$. Then $\alpha_V$ is an isomorphism. For naturality, we have for a map $\phi: V\to W$ that
    \begin{align*}
        \phi(\alpha_V(e_i)) &=\phi(v_i) \\
        &= \sum_j (G\phi)_{i,j} w_j \\
        &= \alpha_W(\sum_j(G\phi)_{i,j}e_j) \\
        &=\alpha_W (FG \phi)(e_i).
    \end{align*}
    In the other direction, $GF$ is the identity functor, which completes the proof.
    \end{proof}
\end{exer}

\begin{exer}[1.2.A] Show that any two initial objects are uniquely isomorphic, and any two final objects are uniquely isomorphic.
    \begin{proof}
    Let $i_1, i_2$ be initial. Then there is exactly one map $f_{12}: i_1 \to i_2$ and likewise $f_{21}: i_2 \to i_1$. Since there is also exactly one map from $i_1$ to itself, the identity, so because $f_{21}\circ f_{12}$ is one such map, it must be the identity. Likewise $f_{12}\circ f_{21}$ is the identity, which yields the unique isomorphisms. The argument is exactly the same for final objects.
    \end{proof}
\end{exer}
\begin{exer}[1.2.B]
What are the initial and final objects (if they exist) in $\catname{Ring}$, $\catname{Set}$, and $\catname{Top}$, or the category of subsets of a set under inclusion, or the category of open subsets of a topological space under inclusion?
\begin{soln}
In $\catname{Ring}$, the initial object is the integers, since $1$ mapping to $1$ dictates where the remaining integers go; the terminal object is the zero ring, since everything must map to zero. In $\catname{Set}$, the initial object is the empty set; the final object is the singleton. In $\catname{Top}$, the initial object is the empty space; the final object is the singleton. In the category of subsets or open sets, the initial object is the empty set/space; the terminal object is the entire set/space.
\end{soln}
\end{exer}

\begin{exer}[1.2.C]
    Let $A$ be a commutative ring and $S$ a multiplicative subset of $A$. Show that $f: A \mapsto S^{-1}A$ via $a \mapsto a/1$ is injective if and only if $S$ contains no zerodivisors.
    \begin{proof}
    First suppose $f$ is not injective, i.e. there is a nonzero element $a \in A$ such that $a/1 = 0/1$. So there exists some $s \in S$ such that $s(a1 - 01) = sa = 0$. Then $s$ is a zerodivisor. 

    Conversely, suppose $S$ contains a zerodivisor $s$, with $st = 0$ for $t$ nonzero. Then $0 = st = s(t1-01)$, $t/1 = 0/1 =0$, so $f(t) = t/1 = 0/1$ and $f$ fails to be injective.
    \end{proof}
\end{exer}

\begin{exer}[1.2.D] Verify that $\iota: A\to S^{-1}A$ is initial among $A$-algebras $B$ where every element of $S$ is sent to an invertible element in $B$, i.e. any map $\phi: A\to B$ sending all elements of $S$ to invertible elements in $B$ factors through $S^{-1}A$.
    \begin{proof}
        Define $\tilde{\phi}: S^{-1}A  \to B$ via $\tilde{\phi}(a/s) := \phi(a)\phi(s)^{-1}$. This makes sense since we have assumed that $\phi(s)$ is invertible. Let $a, b \in A$ and $s,t \in S$. Then 
        \[\tilde{\phi}(ab/st) = \phi(ab)\phi(st)^{-1} = \phi(a)\phi(b)\phi(s)^{-1}\phi(t)^{-1} = \tilde{\phi}(a/s) \tilde{\phi}(b/t)\]
        and 
        \begin{align*}
        \tilde{\phi}(a/s + b/t) &= \tilde{\phi}((at+sb)/st) \\
        &= \phi(at+sb)\phi(st)^{-1} \\
        &= \phi(a)\phi(t)/\phi(s)\phi(t)+\phi(s)\phi(b)/\phi(s)\phi(t) \\
        &= \tilde{\phi}(a/s) + \tilde{\phi}(b/t).
        \end{align*}
        so $\tilde{\phi}$ is a ring homomorphism. And $\tilde{\phi}(\iota(a)) = \tilde{\phi}(a/1) = \phi(a)/\phi(1) = \phi(a)$ as desired. 
    \end{proof}
\end{exer}

\begin{exer}[1.2.E] The localization of an $A$-module $M$ defined via the universal property, actually exists.
    \begin{proof}
        Following the hint, define $S^{-1}M$ to have elements of the form $m/s$ where $m \in M$ and $s \in S$, subject to the equivalence relation where $m/s \sim m'/t$ if there exists an element $s' \in S$ such that $s'(ms - m't) = 0$, with the standard operation and $S^{-1}A$-module operation. Define $\iota: M\to S^{-1}M$ via $m \mapsto m/1$. 

        Let $\phi: M \to N$ be an $A$-module map such that multiplication by any element of $s$ is an isomorphism $N\to N$. Then we can define a map $\tilde{\phi}: S^{-1}M\to N$ via $m/s \mapsto s^{-1}\phi(m)$, which makes sense taking $s^{-1}$ to be the inverse to multiplication by $s$. This is an $A$-module map and $\tilde{\phi}(\iota(m)) = \tilde{\phi}(a/1) = 1^{-1}\phi(a) = \phi(a)$.
    \end{proof}
\end{exer}

\begin{exer}[1.2.F] Properties of localization.
    \begin{enumerate}[(a)]
        \item Localization commutes with finite products, i.e. for $A$-modules $M_1, ..., M_n$ provide an $A$-module isomorphism $S^{-1}(M_1\times \cdots M_n) \to S^{-1}M_1\times \cdots \times S^{-1}M_n$. 
        \begin{proof}
            Define our map via $(m_1, ..., m_n)/s\mapsto (m_1/s, ..., m_n/s)$. For surjectivity, given an element $(m_1/s_1, ..., m_n/s_n)$, the element
            \[(s_2\cdots s_nm_1, ..., s_1 \cdots s_{n-1}m_n)/s_1\cdots s_n\]
            will map to it. For injectivity, suppose $(m_1/s, ..., m_n/s)$ is zero, i.e. there exist elements $t_i \in S$ such that $t_im_i = 0$ for all $i$. Then  $t_1\cdots t_n(m_1, ..., m_n) = 0$, so $(m_1, ..., m_n)/s = 0$.
        \end{proof}
        \item Localization commutes with arbitrary direct sums.
        \begin{proof}
            Let $\Lambda$ be an indexing set. Define a map $S^{-1} (\bigoplus_{\lambda \in \Lambda} M_\lambda) \to \bigoplus_{\lambda \in \Lambda} S^{-1}M_{\lambda}$ via 
            \[(m_\lambda)_{\lambda \in \Lambda}/s \mapsto (m_\lambda/s)_{\lambda \in \Lambda}.\]
            For surjectivity, given an element $(m_\lambda / s_\lambda)_{\lambda \in \Lambda}$, we know that only finitely many indices are nonzero, say $\lambda_1, ..., \lambda_k$. Then let $s= \lambda_1\cdots \lambda_k$, and consider the element $(k_\lambda m_\lambda)_{\lambda \in \Lambda}$ where $k_{\lambda_i} = s/s_i$ for the nonzero indices and zero otherwise. Then it maps to what we want. For injectivity, suppose $(m_\lambda/s)_{\lambda \in \Lambda}$ is zero, i.e. there exist elements $t_\lambda$ such that $t_\lambda m_\lambda = 0$ for all $\lambda$. Again only finitely many of the $m_\lambda$ are nonzero, so take the corresponding $t_\lambda$ and let their product be $t$. Then $t(m_\lambda)_{\lambda \in \Lambda} = 0$ and we are done.
        \end{proof}
        \item Localization does not commute with infinite products, i.e. the same map defined as above is not always an isomorphism for infinite products. 
        \begin{proof}
            We know that $\Z$ localized at $\Z\setminus \{0\}$ is $\Q$, so that $\prod_{\N}\Q$ contains for example the element $x:= (1, 1/2, 1/3, ...)$. On the other hand, suppose there were an element in $(\Z\setminus \{0\})^{-1} (\Z^\N)$ mapping to $x$, i.e. some element $(a_1, a_2, ...)/s$ where $a_i \in \Z$ and $s \in \Z\setminus \{0\}$ such that $a_n/s = 1/n$ for all $n \in \N$. Then $na_n = s$ for all $n \in \N$, i.e. $s$ is an integer divisible by all natural numbers, a contradiction.
        \end{proof}
    \end{enumerate}
\end{exer}

\begin{exer}[1.2.G] Show that $\Z/10 \tensor \Z/12 \cong \Z/2$.
    \begin{proof}
        We define a map $\tilde{f}: \Z/10 \times \Z/12 \to \Z/2$ via $([a],[b]) \mapsto [ab]$. This is clearly bilinear. If $a=a'+10n$ and $b  = b'+12m$, so $[ab] = [a'b'+12ma'+10nb'+120nm] = [a'b']$ in $\Z/2$, so this map is well defined. By the universal property of tensor products, $f$ factors through the tensor product to yield a map $f: \Z/10 \tensor \Z/12\to \Z/2$ such that $[a]\tensor [b] \mapsto [ab]$.

        Consider the map $g: \Z/2 \to \Z/10 \tensor \Z/12$ given by $[a] \mapsto [a]\tensor [1]$. This is again bilinear, and if $a = a' + 2n$, then $[a] \tensor [1] = [a'+2n]\tensor [1] = [a'] \tensor [1] + [12n-10n]\tensor [1] = [a'] \tensor [1] + [1]\tensor [12n] - [10n]\tensor [1] = [a']$, so this is well defined.

        We moreover have $fg([a]) =  f([a]\tensor [1]) = [a]$ and $gf([a],[b]) = g([ab]) = [ab]\tensor [1] = [a]\tensor [b]$. Hence $f$ is an isomorphism. (Note that this argument works to show in general that $\Z/m \tensor \Z/n \cong \Z/\gcd(m,n))$.
    \end{proof}
\end{exer}

\begin{exer}[1.2.H] Show that $(-)\tensor_A N$ is a covariant functor $\catname{Mod}_A \to \catname{Mod}_A$, and that it is right exact.
    \begin{proof}
    Let $f: M\to M'$ be an $A$-module map. Then $(m,n) \to f(m)\tensor n$ is bilinear, so it yields a map $f_*: M\tensor N\to M'\tensor N$, which on generators is given by $m\tensor n\mapsto f(m)\tensor n$. This respects composition, thus making $(-)\tensor N$ into a functor.

    Let $M\xrightarrow{f} M' \xrightarrow{g} M'' \xrightarrow{h} 0$ be exact. Tensoring, we get a complex 
    \[M \tensor N \xrightarrow{f_*} M'\tensor N\xrightarrow{g_*} M''\tensor N \xrightarrow{h_*} 0,\]
    which we need to check is exact.  We know by the first isomorphism theorem that $g$ induces an isomorphism $ M'/\im(f)\to M''$. 
    \begin{claim}
    Tensor products commute with quotients.
    \begin{proof}
        Let $P, Q, R$ be $A$-modules. Suppose $Q$ is a submodule of $P$. Consider the map $\tilde{f}: (P/Q)\times R\to (P\tensor R)/(Q\tensor R)$ given by $([p], r) \mapsto [p\tensor r]$. If $[p]= [p']$, then $[p\tensor r] - [p'\tensor r] = [p-p'\tensor r] = [q \tensor r] = 0$ so this map is well defined. One can also check bilinearity. Hence it extends to a map $f: (P/Q)\tensor R\to(P\tensor R)/(Q\tensor R)$.

        We now define a map $\tilde{g}: P\times R\to (P/Q)\tensor R$ via $(p,r) \mapsto [p]\tensor r$. This is bilinear, hence yields a map $g': P\tensor R \to  (P/Q)\tensor R$, which further factors through the quotient by $Q\tensor R$, yielding a map $g: (P\tensor R)/(Q\tensor R)\to (P/Q)\tensor R$. And $f,g$ are mutually inverse, yielding the claim.
    \end{proof}
    \end{claim}
    By the claim, $g_*$ induces an isomorphism (since functors send isos to isos)
    \[M''\tensor N \leftarrow (M'/\im(f)) \tensor N \cong (M'\tensor N)/\im(f_*)\] 
    so $\im(f_*) = \ker(g_*)$. And surjectivity of $g_*$ follows directly from surjectivity of $g$. 
    \end{proof}

\end{exer}
\begin{exer}[1.2.I]
The tensor product $(T,t)$ of two modules $M, N$ is unique up to unique isomorphism. (Here $t: M\times N \to T$ is the bilinear map given by the universal property.)
\begin{proof}
    Let $(T', t')$ be another pair satisfying the universal property. Then $t'$ must factor through $T$, i.e. there exists a unique map $f: T\to T'$ such that $f\circ t = t'$. Likewise, there exists a unique map $g: T' \to T$ such that $g\circ t' = t$. Plugging these two formulas into each other we get $f\circ g\circ t' =t'$ and $g\circ f \circ t = t$.

    Now we use the universal property again. Since $t'$ is a bilinear map from $M\times N$ to $T'$, it must factor uniquely through $T'$. One such factoring is as above, $(f\circ g)\circ t' =t'$, but another factoring is $\id_{T'} \circ t'$, so by uniqueness, $f\circ g = \id_{T'}$. Likewise, $g\circ f = \id_{T}$, hence $f, g$ are inverses, and the unique isomorphisms between $T, T'$.
\end{proof}
\end{exer}

\begin{exer}[1.2.J]
Show that the tensor product actually exists, i.e. the construction from $\S 1.2.5$ satisfies the universal property.
\begin{proof}
Recall that $M\tensor N$ is the free $A$-module generated by $M\times N$, quotiented by the following relations:
\begin{enumerate}[(a)]
    \item $ (m_1 + m_2, n) - (m_1, n) - (m_2, n) $
    \item  $(m, n_1+n_2)- (m, n_1)- (m, n_2)$
    \item  $a(m, n)- (am, n) $
    \item $ a(m, n)- (m, an)$
\end{enumerate}
for $a \in A, m, m_1, m_2 \in M, n, n_1, n_2 \in N$. Let $f: M\times N \to P$ be a bilinear $A$-module morphism. It follows by bilinearity that 
\begin{enumerate}[(a)]
    \item $f((m_1 + m_2, n) - (m_1, n) - (m_2, n))  = f(m_1, n)  + f(m_2, n) - f(m_1, n) - f(m_2, n)  =  0$
    \item $f((m, n_1+n_2)- (m, n_1)- (m, n_2)) = f(m,n_1) + f(m,n_2) - f(m,n_1)-f(m,n_2) = 0$
    \item $f(a(m, n)- (am, n)) = af(m,n)-af(m,n) = 0$
    \item $f(a(m, n)- (m, an)) = af(m,n) - af(m,n) = 0$
\end{enumerate}
so $f$ factors uniquely through the quotient.
\end{proof}
\end{exer}

\begin{exer}[1.2.K] ``Important exercise''
\begin{enumerate}[(a)]
    \item If $M$ is an $A$-module and $\phi: A\to B$ is a ring morphism, give $B\tensor_A M$ the structure of a $B$-module. Show that this gives a functor $\catname{Mod}_A\to \catname{Mod}_B$.
    \begin{proof}
        Regard $B$ as an $A$-module via the action $A \times B\to B: (a,b) \mapsto \phi(a)b$. Now $B\tensor_A M$ makes sense as an $A$-module. Now for fixed $b\in B$, consider the map $\tilde{f}_b: B\times M\to B\tensor_A M$ given by $(b', m)\to bb'\tensor m$. We compute
        \[\tilde{f}_b(\phi(a)b', m) = b\phi(a)b'\tensor m = \phi(a)(bb'\tensor m) = a\cdot\tilde{f}_b(b',m)\]
        and 
        \[\tilde{f}_b(b' ,am) = bb'\tensor am = a\cdot (bb'\tensor m) = a\cdot \tilde{f}_b(b',m)\]
        so $\tilde{f}_b$ is $A$-bilinear, and it induces a map $f_b: B\tensor_A M \to B \tensor_A M$, which yields the $B$-module structure on $B\tensor_A M$.

        To see that this is a functor, suppose we have an $A$-module morphism $g: M\to N$. Then we get a map $B\times M \to B\tensor_A N$ given by $(b,m)\mapsto b\tensor g(m)$. This is again $A$-bilinear, so it induces a map $g_*: B\tensor_A M\to B \tensor_A N$, which acts on elementary tensors via $b\tensor m \mapsto b\tensor g(m)$.
    \end{proof}
    \item Following notation above, if $\psi: A\to C$ is a ring morphism, then $B\tensor_A C$ has a ring structure.
    \begin{proof}
        Again, $C$ becomes an $A$-module via the action $(a,c) \mapsto \psi(a)c$. Then $B \tensor_A C$ automatically gets an abelian group structure. To define the ring product, note that specifying a map
        \[(B\tensor C)\times (B\tensor C) \to B\tensor C\]
        is the same thing as specifying a map 
        \[B\tensor C \to \hom_{\catname{Mod}_A}(B\tensor C, B\tensor C)\]
        so we can start here. First define a map 
        \[B\times C \to \hom_{\catname{Mod}_A}(B\times C, B\tensor C):  (b_1,c_1) \mapsto ((b_2,c_2)\mapsto b_1b_2 \tensor c_1c_2).\]
        By properties of the tensor product, the image of this map is an $A$-bilinear map, so we get a map 
        \[B\times C \to \hom_{\catname{Mod}_A}(B\tensor C, B\tensor C): b_1, c_1 \mapsto (b_2\tensor c_2 \mapsto b_1b_2 \tensor c_1c_2).\]
        Again, this is also $A$-bilinear. For example we can check 
        \[\phi(a)b_1 \tensor c_1 \mapsto (b_2\tensor c_2 \mapsto \phi(a)b_1b_2 \tensor c_1c_2 ) = (b_2\tensor c_2 \mapsto \phi(a)(b_1b_2 \tensor c_1c_2)).\]
        Hence we get a map 
        \[B\tensor C \to \hom_{\catname{Mod}_A}(B\tensor C, B\tensor C):  b_1\tensor c_1 \mapsto (b_2\tensor c_2 \mapsto b_1b_2 \tensor c_1c_2)\]
        which is equivalent to the map 
        \[\mu: (B\tensor C)\times (B\tensor C) \to B\tensor C: (b_1\tensor c_1), (b_2\tensor c_2)\mapsto b_1b_2\tensor  c_1c_2\]
        which is $A$-bilinear by definition since it comes from the universal property. Now for $x, y, z \in B\tensor C$, we have by bilinearity
        \[\mu(x+y, z) = \mu(x, z)+ \mu(y,z), \quad \mu(x,y+z) = \mu(x,y)+\mu(x,z)\]
        so this satisfies distributivity. The identity element is $1_B\tensor 1_C$, and $\mu$ is obviously associative, so we are done.
    \end{proof}
\end{enumerate}
\end{exer}

\begin{exer}[1.2.L] If $S$ is a multiplicative subset of $A$ and $M$ is an $A$-module, describe a natural isomorphism $(S^{-1}A)\tensor_A M\to S^{-1}M$ (as both $A$-modules and $S^{-1}A$-modules).
    \begin{proof}
    Consider first the map $S^{-1}A\times M \to S^{-1}M$ given by $(a/s, m)\mapsto  (a\cdot m)/ s$. This is $A$-bilinear since 
    \[(a'\cdot a/s, m) = (a'a/s, m) \mapsto (aa'\cdot m)/s = a\cdot (a'\cdot m)/s\]
    and 
    \[(a/s, a'\cdot m) \mapsto (a\cdot(a'\cdot m))/s = (aa'\cdot m)/s = a\cdot (a'\cdot m)/s.\]
    Hence this induces a map $f: (S^{-1}A)\tensor_A M \to S^{-1}M$ which acts on elementary tensors as 
    \[f(a/s \tensor m) = (a\cdot m)/s.\]

    We can define an inverse map as follows. First consider the map $\tilde{g}:M\to (S^{-1}A)\tensor_A M$ via $m \mapsto 1/1 \tensor m$. Since multiplication by elements of $S$ are invertible in $(S^{-1}A)\tensor_A M$, the map $\tilde{g}$ factors through the localization to yield a map $g: S^{-1}M\to (S^{-1}A)\tensor_A M$ which acts via $m/s \mapsto s^{-1} (1/1\tensor m) = (1/s\tensor m)$. Then we can check 
    \[gf(a/s\tensor m) = g((a\cdot m)/s) = (1/s \tensor a\cdot m) = (a/s \tensor m)\]
    and 
    \[fg(m/s) = f(1/s\tensor m) = m/s.\]

\begin{exer}[1.2.M] Show that tensor products commute with arbitrary direct sums, i.e. for $A$-modules $M$ and $\{N_\alpha\}_{\alpha \in \Lambda}$, define an isomorphism $M \tensor (\oplus_{\alpha \in \Lambda} N_{\alpha}) \to \oplus_{\alpha \in \Lambda}(M\tensor N_{\alpha})$.
    \begin{proof}
    As with every map out of a tensor prouduct, we begin by defining a map 
    \[\tilde{f}: M \times (\oplus_{\alpha \in \Lambda} N_{\alpha}) \to \oplus_{\alpha \in \Lambda}(M\tensor N_{\alpha}), (m, (n_\alpha)_{\alpha \in \Lambda}) \mapsto (m\tensor n_\alpha)_{\alpha \in \Lambda}.\]
    This, being bilinear, induces a map $f: M\tensor (\oplus_{\alpha \in \Lambda} N_{\alpha}) \to \oplus_{\alpha \in \Lambda}(M\tensor N_{\alpha})$ which acts on elementary tensors via $m\tensor (n_\alpha)_{\alpha \in \Lambda}\mapsto (m\tensor n_\alpha)_{\alpha \in \Lambda}$.

    For each $\alpha \in \Lambda$, consider the map 
    \[\tilde{g}_\alpha: M\times N_{\alpha}\to M\tensor (\oplus_{\alpha \in \Lambda} N_\alpha), (m, n) \mapsto m\tensor \iota_\alpha(n)\]
    where $\iota_\alpha: N_\alpha \to \oplus_{\alpha}N_\alpha$ is the inclusion. This is $A$-bilinear, so it induces a map $g_\alpha: M\tensor N_\alpha \to M\tensor (\oplus_{\alpha \in \Lambda} N_{\alpha})$. The sum of all the $g_\alpha$ yields a map $g:= \oplus_{\alpha \in \Lambda}(M\tensor N_\alpha) \to M\tensor (\oplus_{\alpha \in \Lambda} N_\alpha)$.

    And we check 
    \begin{align*}
    fg((m_\alpha \tensor n_\alpha)_{\alpha \in \Lambda}) &= f\sum_{\alpha \in \Lambda} (m_\alpha \tensor \iota_\alpha(n_\alpha)) \\
    &= \sum_{\alpha \in \Lambda}f(m_\alpha\tensor \iota(n_\alpha)) \\
    &= \sum_{\alpha \in \Lambda} (..., 0, m_\alpha\tensor n_\alpha, 0, ...) \\
    &= (m_\alpha \tensor n_\alpha)_{\alpha \in \Lambda}
    \end{align*}
    and 
    \[gf( m \tensor (n_\alpha)_{\alpha\in \Lambda}) = g((m\tensor n_\alpha)_{\alpha \in \Lambda}) = \sum_{\alpha \in \Lambda} m\tensor \iota_\alpha(n_\alpha) = m\tensor \sum_{\alpha \in \Lambda} \iota_\alpha(n_\alpha) = m\tensor (n_\alpha)_{\alpha \in \Lambda}\]
    so $f$ is an isomorphism.
    \end{proof}
\end{exer}

\begin{exer}[1.2.H] Show that in $\catname{Set}$, the fiber product is given by $X\times_Z Y = \{(x,y) \in X\times Y: \alpha(x) = \beta(y)\}$.
    \begin{proof}
        Call the right hand side of above $R$.
        Let $W$ be a set with maps $f_X: W\to X$ and $f_Y: W\to Y$ such that $\alpha\circ f_X = \beta\circ f_Y$. Define the map $\phi: W\to R$ via $\phi(w) = (f_X(w), f_Y(w))$. This actually lands in $R$ since $\alpha\circ \pi_X(\phi(w)) = \alpha \circ f_X(w) = \beta \circ f_Y(w) = \beta \circ \pi_Y(\phi(w))$.  Then clearly $\pi_X \circ \phi = f_X$, and $\pi_Y \circ \phi = f_Y$. Thus $R$ satisfies the universal property so it is \emph{the} fiber product in the category of sets.
    \end{proof}
\end{exer}

\begin{exer}[1.2.O]
If $X$ is a topological space, show that fiber products always exist in the category of open sets of $X$ by describing what a fibered product is.
\begin{proof}
    Remember that the morphisms here are just inclusions of open sets. So suppose we have open sets $U, U'$ mapping into $W$, i.e. $U, U'\subset W$. We claim that the fiber product here is just $U\cap U'$ (intersections of two open sets are open), with the morphisms given by inclusions into $U, U'$ . To see this, given another open set $V$ with morphisms $V\to U$ and $V\to U'$ that agree with $U \to W$ and $U' \to W$, then we know that $V$ must be a subset of both $U$ and $U'$. It follows that $V$ is a subset of $U\cap U'$. And by definition, all the morphisms are inclusions, so everything commutes.
\end{proof}
\end{exer}

\begin{exer}[1.2.T] Show that the coproduct in the category of sets is disjoint union.
    \begin{proof}
    Let $X, Y$ be sets and let us take the disjoint union to be 
    \[X\coprod Y:= \{(x,0): x \in X\}\cup \{(y,1): y \in Y\}.\]
    Define the inclusions $\iota_X: X\to X\coprod Y$ via $x \mapsto (x,0)$ and $\iota_Y: Y\to X\coprod Y$ via $y\mapsto (y,1)$.
    Let $A$ be a set with maps $f_X: X\to A$, $f_Y:Y\to A$. Then we can define a map $f: X\coprod Y \to A$ via $(x,0)\mapsto f_X(x)$ and $(y,1)\mapsto f_Y(y)$, which makes everything commute. For uniqueness, if $g: X\coprod Y\to A$ is another map making things commute, we have 
    \[g(x,0) = g(\iota_X(x)) = f_X(x) = f(x,0)\]
    and likewise for $Y$.
    \end{proof}
\end{exer}

\end{tikzcd}\]
\begin{proof}
    To see that $b\to b\tensor 1$ is a ring map, we have $b+b' \mapsto (b+b')\tensor 1 = b\tensor 1 + b'\tensor 1$, and $bb' \mapsto bb' \tensor 1 =( b\tensor 1 )(b'\tensor 1)$, and $1\mapsto 1\tensor 1$. Likewise for $C\to B\tensor_A C$. The square commutes since $f_B(a)\tensor 1 = f_B(a)\cdot 1 \tensor 1 = 1\tensor f_C(a) \cdot 1 = 1\tensor f_C(a)$.

    Let $D$ be another ring with maps $g_B: B\to D$ and $g_C: C\to D$ such that $g_Bf_b = g_Cf_C$. Define a map 
    $B\tensor_A C\to D$ beginning with $B\times C\to D, (b, c)\mapsto g_B(b)g_C(c)$. By virtue of $g_B$ being a ring morphism, we have 
    \[(f_B(a)b, c)\mapsto g_B(f_B(a)b)g_C(c) = g_B(f_B(a))g_B(b)g_C(c)\]
    and likewise in the second argument, so this map is $A$-bilinear. Consequently we have a map $\phi: B\tensor_A C\to D$ acting on basis elements via $b\tensor c\mapsto g_B(b)g_C(c)$. Then $g_B(b) = g_B(b)1 = g_B(b)g_C(1) =\phi(b\tensor 1)$, and likewise $g_C(c) = \phi(1\tensor c)$, so we are done.
\end{proof}
\end{exer}

\begin{exer}[1.2.V]
Show that the composition of two monomorphisms is a monomorphism.
\begin{proof}
Let $X\xrightarrow{f} Y\xrightarrow{g} Z$ be monomorphisms. Suppose $\pi_1, \pi_2: W\to X$ satisfy $(g\circ f) \circ \pi_1 = (g\circ f)\circ \pi_2$. Then by associativity of composition, we have $g \circ (f\circ \pi_1) = g\circ(f\circ \pi_2)$, and because $g$ is a mono, $f\circ \pi_1 = f\circ \pi_2$, and because $f$ is a mono, $\pi_1 = \pi_2$.
\end{proof}
\end{exer}

\begin{exer}[1.2.W]
Show that a morphism $\pi: X\to Y$ is a monomorphism if and only if the fiber product $X\times_Y X$ exists and the induced diagonal map $\delta_\pi: X\to X\times_Y X$ is an isomorphism.
\begin{proof}
Forward direction. We claim that the fiber product is given by any object $X'$ with a chosen isomorphism $\psi: X' \to X$ serving as both projection maps. This exists, for example, by taking $X'=X$ and $\psi = \id$. To see this is the fiber product, it is clear that the square commutes. Given any maps $f_1, f_2: W\to X$ such that $\pi\circ f_1 = \pi \circ f_2$, we know that $f_1 = f_2$ since $\pi$ is a monomorphism. Then defining $W\to X$ by $\psi^{-1} \circ f_1$ makes things commute. If $(X'', \psi'')$ is another fiber product, then $\psi''^{-1}\circ \psi$ gives the unique isomorphism, so $(X', \psi)$ satisfies the universal property. Finally, the diagonal map must satisfy $\psi \circ \delta_\pi = \id$, so it is an isomorphism since $\psi$ is an isomorphism.

Backward direction. By assumption, $\delta_\pi$ is an isomorphism, and it satisfies the identity $\mathrm{pr}_X \circ \delta_\pi = \id$, so both projection maps $\mathrm{pr}_X$ are inverse to $\delta_\pi$, so both projection maps are the same. Then given maps $f_1, f_2: W\to X$ such that $\pi \circ f_1 = \pi \circ f_2$, by the the universal property of fiber products we get a unique map $\phi: W\to X\times_Y X$ such that $f_1 = \delta_\pi^{-1}\circ \phi = f_2$.
\end{proof}
\end{exer}

\begin{exer}[1.2.Y] Yoneda lemma.
    \begin{enumerate}[(a)]

            \[\begin{tikzcd}[cramped]

\begin{exer}[1.3.A] Show that if $\mathcal{I}$ has an initial object $e$, then the limit of any diagram $F: \mathcal{I}\to \mathcal{C}$ exists.
    \begin{proof}
    We claim that the limit is just given by $Fe$, with the structure maps $f_j: Fe \to Fj$ given by the image of the unique morphism $e \to j$ under $F$, say $g_j$.     \end{proof}
\end{exer}

\begin{exer}[1.3.C] In the category of sets, the limit of a diagram $F: \mathcal{I} \to A$ is given by 
    \[L:= \{(a_i)_{i \in \mathcal{I}} \in \prod_{i\in I}A_i: F(m)(a_j) = a_k \text{ for all $m \in \Hom(j,k)$}\}\]
    with the standard projection maps $\pi_i: L \to A_i$.
    \begin{proof}
        By definition of $L$, given $a= (a_i)_{i\in \mathcal{I}} \in L$ and $m: j\to k$, we have by definition of $L$ that $Fm(\pi_j(a)) = Fm(a_j) = a_k = \pi_k(a)$. Hence things commute. To show that $L$ is final, suppose we have a family of functions $g_i: W\to A_i$ commuting with things. Then we can define $g: W\to L$ given by $w \mapsto (g_i(w))_{i \in \mathcal{I}}$. Then $\pi_j(g(w)) = g_j(w)$ for $j \in \mathcal{I}$, completing the proof.
    \end{proof}
\end{exer}

\begin{exer}[1.3.D] Colimits.
    \begin{enumerate}[(a)]
        \item Interpret the statement $\Q = \colim (\frac{1}{n}\Z)$.
        \begin{soln}
            Consider the category of natural numbers $\N$ with morphisms given by divisibility, i.e. there is a morphism $a\to b$ if $a$ divides $b$. Consider the functor $F$ from $\N$ to abelian groups given by $n\mapsto \frac{1}{n}\Z$. Then if there is an arrow $n\to m$, i.e. $nk=m$ for some $k$, we define a homomorphism $\phi_{nm}: \frac{1}{n}\Z\to\frac{1}{m}\Z$ via $x/n\mapsto xk/m$. Thus $F$ is actually a functor, and we claim that $\Q$ is the colimit of this diagram.

            To see this, note that we have maps $\iota_n: \frac{1}{n}\Z \to \Q$ given by inclusion. These commute with the morphisms in $\N$, again with $n,m,k$ as above we have $\iota_n(x/n) = x/n = xk/nk = xk/m =\iota_m(\phi_{nm}(x/n))$. Given another group $K$ with morphisms $g_n: \frac{1}{n}\Z \to K$ for all $n$ commuting with the divisibility maps, we can define a map $\phi: \Q\to K$ via $q=a/b \mapsto g_b(q)$. 
            
            This is well defined, since if $a/b = a'/b'$, then the following diagram commutes
            \[
            \begin{tikzcd}[cramped]
                % https://q.uiver.app/#q=WzAsNSxbMCwwLCJcXGZyYWN7MX17Yn1cXFoiXSxbMCwxXSxbMCwyLCJcXGZyYWN7MX17Yid9XFxaIl0sWzEsMSwiXFxmcmFjezF9e2JiJ31cXFoiXSxbMywxLCJLIl0sWzAsMywiYS9iIFxcbWFwc3RvIGFiJy9iYiciXSxbMiwzLCJhJy9iJ1xcbWFwc3RvIGEnYi9iYiciLDJdLFszLDQsImdfe2JiJ30iLDJdLFswLDQsImdfYiIsMCx7ImN1cnZlIjotMn1dLFsyLDQsImdfe2InfSIsMix7ImN1cnZlIjoyfV1d

                {} & {\frac{1}{bb'}\Z} && K \\
                {\frac{1}{b'}\Z}
                \arrow["{a/b \mapsto ab'/bb'}", from=1-1, to=2-2]
                \arrow["{g_b}", curve={height=-12pt}, from=1-1, to=2-4]
                \arrow["{g_{bb'}}"', from=2-2, to=2-4]
                \arrow["{a'/b'\mapsto a'b/bb'}"', from=3-1, to=2-2]
                \arrow["{g_{b'}}"', curve={height=12pt}, from=3-1, to=2-4]
            \end{tikzcd}\]
            implying that $\phi(a/b) = g_b(a/b) = g_{b'}(a'/b') = \phi(a'/b')$. And $\phi$ trivially commutes with the $\iota_n$ since we have $\phi(\iota_b(a/b)) = \phi(a/b) = g_b(a/b)$ by definition.
        \end{soln}
        \item Interpret the union of subsets of a given set as a colimit.
        \begin{soln}
            Let $A$ be a set and consider the category of subsets $2^A$ of $A$, with a morphism $X\to Y$ if $X\subset Y$. Let $\{A_i\}_{i \in I}$ be the collection of subsets we want to take the union of, interpreted as a subcategory of the category of subsets of $A$. Then we get a diagram from this subcategory to $2^A$ via inclusion, and we claim that $\cup A_i$ is the colimit of this diagram. The commutativity is obvious, if $A_j\subset A_k$, then $A_j \subset A_k \subset \cup A_i$ gives the same thing as $A_j \subset \cup A_i$. 
            If we have another subset of $A$, say $K$ with maps from each $A_i$ commuting with inclusion, this means that $\cup A_i$ is a subset of $K$, automatically giving us the desired extension. Again all the morphisms commute since they are just inclusions.
        \end{soln}
    \end{enumerate}
\end{exer}

\begin{exer}[1.3.E] Let $\mathcal{I}$ be a filtered index category. Prove that any colimit $F: \mathcal{I} \to \catname{Set}$ exists, and is given by
    \[K:= \big\{(a_i, i) \in \coprod_{i \in \mathcal{I}} A_i\big\} \big/\sim\]
where the equivalence relation is 
\[\text{
            $(a_i, i)\sim (a_j, j)$ iff there are $f: A_i \to A_k$  and $g: A_j \to A_k$ in the diagram such that $f(a_i) = g(a_j)$.
        }\]
        and inclusions $\iota_i: A_i\to K$ are just the standard $a \mapsto (a, i)$.
        \begin{proof}
        First we need to show that $\sim$ is an equivalence relation. For reflexivity, we have an identity morhpism $A_i \to A_i$, so because $\mathcal{I}$ is filtered we have a set $A_j$ with a morphism $A_i\to A_j$, so $(a_i, i)\sim (a_i, i)$. Symmetry is obvious. For transitivity, suppose $(a_i, i) \sim (a_j, j)$, witnessed by some $A_{ij}$, and $(a_j, j) \sim (a_k, k)$, witnessed by some $A_{jk}$, as in the solid arrows in the diagram below:
        \[\begin{tikzcd}[cramped]
            % https://q.uiver.app/#q=WzAsOCxbMSwwLCJBX2oiXSxbMCwxLCJBX2kiXSxbMSwxLCJBX3tpan0iXSxbMywwLCJBX2oiXSxbMywxLCJBX3tqa30iXSxbNCwxLCJBX2siXSxbMiwyLCJBX2wiXSxbMiwzLCJBX2wnIl0sWzAsMl0sWzEsMl0sWzMsNF0sWzUsNF0sWzIsNiwiIiwwLHsic3R5bGUiOnsiYm9keSI6eyJuYW1lIjoiZGFzaGVkIn19fV0sWzQsNiwiIiwwLHsic3R5bGUiOnsiYm9keSI6eyJuYW1lIjoiZGFzaGVkIn19fV0sWzYsNywiIiwwLHsic3R5bGUiOnsiYm9keSI6eyJuYW1lIjoic3F1aWdnbHkifX19XV0=

            {A_i} & {A_{ij}} && {A_{jk}} & {A_k} \\

            && {A_l'}
            \arrow[from=1-2, to=2-2]
            \arrow[from=1-4, to=2-4]
            \arrow[from=2-1, to=2-2]
            \arrow[dashed, from=2-2, to=3-3]
            \arrow[dashed, from=2-4, to=3-3]
            \arrow[from=2-5, to=2-4]
            \arrow[squiggly, from=3-3, to=4-3]
        \end{tikzcd}\]
        Then the dotted arrows and zigzag arrows come from the fact that $\mathcal{I}$ is filtered, so the compositions $A_i \to A_l'$ and $A_k\to A_l'$ show that $(a_i, i) \sim (a_k, k)$. Hence $\sim$ is an equivalence relation.

        Now we check that that $\iota_i$ commute with the morphisms in the diagram. Suppose $m: j\to k$ is a morphism in $\mathcal{I}$. We need to show that $(a, j) = \iota_j(a) = \iota_k \circ Fm(a) = (Fm(a), k)$ for all $a \in A_j$, which by the equivalence relation amounts to finding some $l \in \mathcal{I}$ and maps $f: A_j\to A_l$ and $g: A_k\to A_l$ such that $f(a) = g(Fm(a))$. Since $\mathcal{I}$ is filtered, there exists some $l'$ with maps $\phi_j: A_j \to A_{l'}$ and $\phi_k: A_k \to A_{l'}$. Now $\phi_{l'}$ and $\phi_k\circ Fm$ are two maps $A_j \to A_{l'}$, so again by filteredness there is some $A_l$ and a map $\phi: A_{l'}\to A_l$ such that $\phi \circ \phi_{l'} = \phi \circ \phi_k\circ Fm$. Then taking $f=\phi\circ \phi_{l'}$ and $g= \phi\circ \phi_k$ works. 

        Finally for the universal property. Given another set $L$ with maps $\psi_i: A_i \to L$ commuting with the diagram, define $h: K\to L$ via $(a_i, i) \mapsto \psi_i(a_i)$. This commutes with the $\iota_i$ trivially, so we just need to check it is well-defined. Suppose $(a_i, i)\sim (a_j,j)$. Then there is some $k$ with maps $f: A_i \to A_k$ and $g: A_j\to A_k$ such that $f(a_i) = g(a_j)$. Since the $\psi_i$ commute with these maps, we get $\psi_i = \psi_k(f(a_i)) = \psi_k(g(a_j)) = \psi_j(a_j)$, completing the proof.
        \end{proof}
\end{exer}
